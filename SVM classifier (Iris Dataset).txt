'''
# A Support Vector Machine (SVM) is a powerful supervised machine learning 
algorithm primarily used for data classification, though it also handles 
regression tasks. The main objective of an SVM classifier is to find the 
optimal hyperplane in an N-dimensional space that maximizes the boundary 
margin between distinct classes.
'''

from sklearn import datasets
from sklearn.model_selection import train_test_split
from sklearn.svm import SVC
from sklearn.metrics import accuracy_score, classification_report

iris = datasets.load_iris()
X = iris.data # Features: sepal length, sepal width, petal length, petal width
y = iris.target # Target labels: setosa, versicolor, virginica
target_names = iris.target_names

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

svm_model = SVC(kernel='linear', random_state=42)
svm_model.fit(X_train, y_train)

y_pred = svm_model.predict(X_test)

print("Accuracy:", accuracy_score(y_test, y_pred))
print("\nClassification Report:\n", classification_report(y_test, y_pred, target_names=target_names))
